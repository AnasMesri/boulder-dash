9   import javax.swing.*;
10  import java.awt.*;
11  import java.util.ArrayList;
12  import java.util.Collection;
13  /**
14   * abstract class Elements
15   * The father of every objects printed in the window
16   * @see JComponent
17   * @see IComponent
18   * @see IAction
19   * */
20  @SuppressWarnings("rawtypes")
21  public abstract class Elements extends JComponent implements IComponent, IAction {
22  
23      /**
24  	 * 
25  	 */
26  	private static final long serialVersionUID = -438201550645736786L;
27  	protected IEngine engine;
28      protected Position position;
29  
30      private Image image;
31  
32      protected SpriteManager spriteManager;
33  
34      
35  	protected Collection<Reaction> reactions = new ArrayList<>();
36      /**
37       * Elements
38       * the constructor of the class. Used to create a new element.
39       * */
40      public Elements(Position position, final Dimension dimension, SpriteManager spriteManager){
41          this.position = position;
42  
43          this.setPreferredSize(dimension);
44          this.setSize(dimension);
45  
46          this.updateBounds();
47  
48          this.spriteManager = spriteManager;
49  
50          this.setImage(this.spriteManager.getImage());
51      }
52  
53      /**
54       * setImage
55       * set the image of the element, like the image passed
56       * @param image image
57       * */
58      public void setImage(Image image) {
59          this.image = image;
60          this.revalidate();
61          this.repaint();
62      }
63  
64      /**
65       * getPosition
66       * return the position of the element
67       * @see Position
68       * @return Position
69       * */
70      public Position getPosition() {
71          return position;
72      }
73      /**
74       * setPosition
75       * configure a new position and set the news bounds of the element
76       * @param position Set a position
77       * */
78      public void setPosition(Position position) {
79          this.position = position;
80          this.updateBounds();
81      }
82  
83      /**
84       * updateBounds
85       * configure the new bounds of the element
86       * @see Rectangle
87       * @see Position
88       * */
89      private void updateBounds(){
90          Rectangle bounds = new Rectangle(this.position.getX(), this.position.getY(), this.getSize().width, this.getSize().height);
91          this.setBounds(bounds);
92      }
93  
94      @Override
95      public boolean isReaction(Object from, Sides side) {
96          for (Reaction reaction: this.reactions){
97              if (reaction.isReaction(from, side)){
98                  return true;
99              }
100         }
101         return false;
102     }
103 
104     @Override
105     public boolean performReaction(Object from, Sides side, int ticks) {
106         for (Reaction reaction: this.reactions){
107             if (reaction.isReaction(from, side)){
108                 return reaction.perform(from, side, ticks);
109             }
110         }
111         return false;
112     }
113 
114     @Override
115     public void paint(Graphics g) {
116         super.paint(g);
117         g.setColor(this.getBackground());
118         Rectangle rec = this.getBounds();
119         g.fillRect(0, 0, rec.width, rec.height);
120         g.drawImage(this.image, 0, 0, this.getWidth(), this.getHeight(), this);
121 
122     }
123 
124     @Override
125     public void setEngine(IEngine engine) {
126         this.engine = engine;
127     }
128 
129     @Override
130     public void explode() {
131 
132     }
133 }
